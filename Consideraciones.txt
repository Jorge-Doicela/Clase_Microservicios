# Explicación de Axios

**Axios** es una librería para hacer peticiones HTTP desde Node.js o el navegador. Es como un mensajero que envía solicitudes a servidores.

## Conceptos básicos:

### 1. **GET** - Obtener datos
```javascript
axios.get('http://api.com/usuarios/1')
  .then(res => console.log(res.data))  // Datos recibidos
  .catch(err => console.log(err));      // Si hay error
```

### 2. **POST** - Enviar datos (crear)
```javascript
axios.post('http://api.com/usuarios', {
  nombre: 'Juan',
  email: 'juan@mail.com'
})
  .then(res => console.log(res.data))
  .catch(err => console.log(err));
```

### 3. **PUT** - Actualizar datos
```javascript
axios.put('http://api.com/usuarios/1', {
  nombre: 'Juan Actualizado'
})
  .then(res => console.log(res.data))
  .catch(err => console.log(err));
```

### 4. **DELETE** - Eliminar datos
```javascript
axios.delete('http://api.com/usuarios/1')
  .then(res => console.log(res.data))
  .catch(err => console.log(err));
```

const service = require('../services/order.service');
exports.getOrders = async (req, res) => {
    const data = await service.getAll();
    res.json(data);
};

exports.getOrderById = async (req, res) => {

};

exports.createOrder = async (req, res) => {
    try {
        if (!req.body || Object.keys(req.body).length === 0) {
            return res.status(400).json({ error: "El cuerpo de la solicitud está vacío" });
        }
        const newOrder = await service.create(req.body);
        res.status(201).json(newOrder);
    } catch (err) {
        res.status(400).json({ error: err.message });
    }
};

exports.updateOrder = async (req, res) => {
    try {
        const updated = await service.update(req.params.id, req.body);
        if (!updated) {
            return res.status(404).json({ error: "Orden no encontrada" });
        }
        res.json(updated);
    } catch (err) {
        res.status(400).json({ error: err.message });
    }
};

exports.deleteOrder = async (req, res) => {
    try {
        const result = await service.delete(req.params.id);
        if (!result) {
            return res.status(404).json({ error: "Orden no encontrada" });
        }
        res.json({ message: "Orden eliminada", orden: result });
    } catch (err) {
        res.status(500).json({ error: err.message });
    }
};
